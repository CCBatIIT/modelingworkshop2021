# Create a dictionary that contains Vina scores for all FDA approved drugs
def getVinaScore(pdbqt_FN):
  F = open(pdbqt_FN,'r')
  line = F.readline()
  while not line.startswith('REMARK VINA RESULT:'):
    line = F.readline()
  F.close()
  score = float(line.split()[3])
  return score
